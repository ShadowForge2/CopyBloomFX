<script>
document.addEventListener('DOMContentLoaded', function() {
    const profilePicWrap = document.getElementById('profile-pic-wrap');
    const fileInput = document.getElementById('id_profile_picture');
    const profilePic = profilePicWrap.querySelector('.profile-pic');
    
    // Handle click on profile picture area
    profilePicWrap.addEventListener('click', function() {
        fileInput.click();
    });
    
    // Handle file selection
    fileInput.addEventListener('change', function(e) {
        const file = e.target.files[0];
        if (file) {
            // Validate file type
            const allowedTypes = ['image/jpeg', 'image/png', 'image/gif', 'image/webp'];
            if (!allowedTypes.includes(file.type)) {
                alert('Please select a valid image file (JPEG, PNG, GIF, or WebP)');
                return;
            }
            
            // Validate file size (5MB)
            if (file.size > 5 * 1024 * 1024) {
                alert('Image size should not exceed 5MB');
                return;
            }
            
            // Show preview
            const reader = new FileReader();
            reader.onload = function(e) {
                profilePic.style.backgroundImage = `url('${e.target.result}')`;
                profilePic.classList.remove('placeholder');
            };
            reader.readAsDataURL(file);
        }
    });
    
    // Handle drag and drop
    profilePicWrap.addEventListener('dragover', function(e) {
        e.preventDefault();
        profilePicWrap.classList.add('drag-over');
    });
    
    profilePicWrap.addEventListener('dragleave', function(e) {
        e.preventDefault();
        profilePicWrap.classList.remove('drag-over');
    });
    
    profilePicWrap.addEventListener('drop', function(e) {
        e.preventDefault();
        profilePicWrap.classList.remove('drag-over');
        
        const files = e.dataTransfer.files;
        if (files.length > 0) {
            fileInput.files = files;
            const event = new Event('change', { bubbles: true });
            fileInput.dispatchEvent(event);
        }
    });
});
</script>
